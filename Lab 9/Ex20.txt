#include <stdio.h>

int negativos(float *vet, int N) {
    int i, negativos = 0;
    for (i = 0; i < N; i++) {
        if (*(vet+i) < 0) {
            negativos++;
        }
    }
    return negativos;
}

int main() {
    float vet[] = {-1.2, 2.5, -3.4, 4.8, -5.6, 6.7};
    int N = sizeof(vet) / sizeof(vet[0]);
    int quantidade = negativos(vet, N);
    printf("Quantidade de nÃºmeros negativos: %d\n", quantidade);
    return 0;
}
