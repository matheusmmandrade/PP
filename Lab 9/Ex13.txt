#include <stdio.h>
#include <string.h>

int ocorre(char *str1, char *str2) {
    int i, j, k, len1, len2;
    len1 = strlen(str1);
    len2 = strlen(str2);
    for (i = 0; i <= len1 - len2; i++) {
        for (j = i, k = 0; str2[k] != '\0' && *(str1 + j) == str2[k]; j++, k++);
        if (k == len2) {
            return 1;
        }
    }
    return 0;
}

int main() {
    char str1[100], str2[100];
    printf("Digite a primeira string: ");
    scanf("%s", str1);
    printf("Digite a segunda string: ");
    scanf("%s", str2);
    if (ocorre(str1, str2)) {
        printf("A segunda string ocorre dentro da primeira.\n");
    }
    else {
        printf("A segunda string nÃ£o ocorre dentro da primeira.\n");
    }
    return 0;
}
