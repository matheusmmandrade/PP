#include <stdio.h>

int max(int *arr, int size) {
    int max = *arr;
    int i;
    for (i = 1; i < size; i++) {
        if (*(arr + i) > max) {
            max = *(arr + i);
        }
    }
    return max;
}

int main() {
    int N, k;
    printf("Informe o tamanho do array: ");
    scanf("%d", &N);
    int arr[N];
    int i;
    printf("Informe os elementos do array: ");
    for (i = 0; i < N; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Informe o nÃºmero de elementos por linha: ");
    scanf("%d", &k);
    printf("Array: ");
    for (i = 0; i < N; i++) {
        printf("%d ", arr[i]);
        if ((i+1) % k == 0) {
            printf("\n");
        }
    }
    printf("\nMaior elemento: %d\n", max(arr, N));
    return 0;
}
